# Q05 Express REST API：完整可跑的 Tasks CRUD（參考解答）

## `app.js`

```js
import express from 'express';

// ---- 自訂錯誤（沿用 Q04）----
class AppError extends Error {
  constructor(message, statusCode = 500) {
    super(message);
    this.statusCode = statusCode;
  }
}

export function createApp() {
  const app = express();
  app.use(express.json()); // 解析 JSON body

  // ---- 假資料層（之後換成 DB）----
  let tasks = [
    { id: 1, title: '寫後端練習', done: false },
    { id: 2, title: '複習 React', done: true },
  ];
  let nextId = 3;

  // ---- 自訂 middleware：簡易請求紀錄（像前端的 console.log，但在伺服器端）----
  app.use((req, res, next) => {
    console.log(`${req.method} ${req.url}`);
    next();
  });

  // ---- 驗證 middleware：檢查 body 是否合法 ----
  function validateTask(req, res, next) {
    const { title } = req.body;
    if (typeof title !== 'string' || title.trim() === '') {
      return next(new AppError('title 為必填且需為非空字串', 400));
    }
    next();
  }

  // ---- 包裝 async handler，讓丟出的錯誤自動進錯誤 middleware ----
  const wrap = (fn) => (req, res, next) => Promise.resolve(fn(req, res, next)).catch(next);

  // GET 全部
  app.get('/tasks', wrap(async (req, res) => {
    res.json(tasks);
  }));

  // GET 單筆
  app.get('/tasks/:id', wrap(async (req, res) => {
    const task = tasks.find((t) => t.id === Number(req.params.id));
    if (!task) throw new AppError('找不到該任務', 404);
    res.json(task);
  }));

  // POST 新增
  app.post('/tasks', validateTask, wrap(async (req, res) => {
    const task = { id: nextId++, title: req.body.title.trim(), done: false };
    tasks.push(task);
    res.status(201).json(task); // 201 Created
  }));

  // PUT 更新
  app.put('/tasks/:id', validateTask, wrap(async (req, res) => {
    const task = tasks.find((t) => t.id === Number(req.params.id));
    if (!task) throw new AppError('找不到該任務', 404);
    task.title = req.body.title.trim();
    if (typeof req.body.done === 'boolean') task.done = req.body.done;
    res.json(task);
  }));

  // DELETE 刪除
  app.delete('/tasks/:id', wrap(async (req, res) => {
    const idx = tasks.findIndex((t) => t.id === Number(req.params.id));
    if (idx === -1) throw new AppError('找不到該任務', 404);
    tasks.splice(idx, 1);
    res.status(204).end(); // 204 No Content
  }));

  // ---- 404 fallback：沒有任何路由匹配 ----
  app.use((req, res, next) => next(new AppError('路由不存在', 404)));

  // ---- 統一錯誤處理 middleware（4 個參數！必須放最後）----
  app.use((err, req, res, next) => {
    const status = err.statusCode || 500;
    if (status === 500) console.error(err); // 真正的伺服器錯誤才記 log
    res.status(status).json({ error: err.message || '伺服器錯誤' });
  });

  return app;
}

// 啟動伺服器（測試時會 import createApp，不會跑這段）
if (process.env.NODE_ENV !== 'test') {
  const app = createApp();
  app.listen(3000, () => console.log('http://localhost:3000'));
}
```

## 驗證

```bash
node app.js
# 另開終端機：
curl http://localhost:3000/tasks
curl -X POST http://localhost:3000/tasks -H 'Content-Type: application/json' -d '{"title":"新任務"}'
curl -X PUT http://localhost:3000/tasks/1 -H 'Content-Type: application/json' -d '{"title":"改過","done":true}'
curl -X DELETE http://localhost:3000/tasks/2
```

## 重點

- **middleware 是有順序的**：`express.json()` 要在路由前、錯誤處理要在最後。
- **`wrap` 樣板很重要**：Express 4 不會自動接住 async handler 丟出的錯誤，少了它 app 會卡住。
- **驗證放 middleware**：把「檢查」和「業務邏輯」分離，乾淨且可重用。
- 狀態碼語意：201 新增成功、204 刪除成功無內容、400 請求格式錯、404 找不到、500 伺服器內部錯。
