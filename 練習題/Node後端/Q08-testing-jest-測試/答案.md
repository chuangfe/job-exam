# Q08 測試（Jest + Supertest）（參考解答）

## `package.json` 設定

```json
{
  "scripts": { "test": "NODE_ENV=test node --experimental-vm-modules node_modules/jest/bin/jest.js" }
}
```

> ES Module + Jest 需要 `--experimental-vm-modules`。若你用 CommonJS，直接 `"test": "jest"` 即可。

## 8-1：單元測試 `validate.test.js`

```js
function isValidTitle(title) {
  return typeof title === 'string' && title.trim().length > 0;
}

describe('isValidTitle', () => {
  test('正常字串通過', () => {
    expect(isValidTitle('寫測試')).toBe(true);
  });
  test('空字串不通過', () => {
    expect(isValidTitle('   ')).toBe(false);
  });
  test('非字串不通過', () => {
    expect(isValidTitle(123)).toBe(false);
    expect(isValidTitle(undefined)).toBe(false);
  });
});
```

## 8-2：整合測試 `app.test.js`

```js
import request from 'supertest';
import { createApp } from './app.js';

describe('Tasks API', () => {
  let app;
  beforeEach(() => {
    app = createApp(); // 每個測試用全新 app，狀態互不污染
  });

  test('GET /tasks 回傳陣列', async () => {
    const res = await request(app).get('/tasks');
    expect(res.status).toBe(200);
    expect(Array.isArray(res.body)).toBe(true);
  });

  test('POST /tasks 新增成功回 201', async () => {
    const res = await request(app).post('/tasks').send({ title: '測試任務' });
    expect(res.status).toBe(201);
    expect(res.body).toMatchObject({ title: '測試任務', done: false });
    expect(res.body.id).toBeDefined();
  });

  test('POST /tasks 缺 title 回 400', async () => {
    const res = await request(app).post('/tasks').send({});
    expect(res.status).toBe(400);
    expect(res.body.error).toBeDefined();
  });

  test('GET 不存在的 task 回 404', async () => {
    const res = await request(app).get('/tasks/99999');
    expect(res.status).toBe(404);
  });

  test('完整流程：建立 → 更新 → 刪除', async () => {
    const created = await request(app).post('/tasks').send({ title: 'A' });
    const id = created.body.id;

    const updated = await request(app).put(`/tasks/${id}`).send({ title: 'B', done: true });
    expect(updated.status).toBe(200);
    expect(updated.body).toMatchObject({ title: 'B', done: true });

    const deleted = await request(app).delete(`/tasks/${id}`);
    expect(deleted.status).toBe(204);

    const after = await request(app).get(`/tasks/${id}`);
    expect(after.status).toBe(404);
  });
});
```

## 跑測試

```bash
npm test
```

## 重點

- **單元測試**測純邏輯（快、穩、好定位）；**整合測試**測「請求進來到回應出去」整條路（更接近真實）。
- 用 `beforeEach` 重建狀態，確保測試彼此獨立（前端測試同理）。
- 測**失敗路徑**（400 / 404 / 401）和成功路徑一樣重要——這是後端工程師的基本功，面試常被問。
