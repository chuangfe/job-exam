## `src/db.js`

```js
import Database from 'better-sqlite3';

const db = new Database(process.env.DB_FILE || ':memory:');

db.exec(`
  CREATE TABLE IF NOT EXISTS users (
    id            INTEGER PRIMARY KEY AUTOINCREMENT,
    email         TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    created_at    TEXT DEFAULT (datetime('now'))
  );

  CREATE TABLE IF NOT EXISTS notes (
    id         INTEGER PRIMARY KEY AUTOINCREMENT,
    user_id    INTEGER NOT NULL,
    title      TEXT NOT NULL,
    content    TEXT DEFAULT '',
    created_at TEXT DEFAULT (datetime('now')),
    updated_at TEXT DEFAULT (datetime('now')),
    FOREIGN KEY (user_id) REFERENCES users(id)
  );
`);

export default db;
```

## `src/repos/noteRepo.js`

```js
import db from '../db.js';

// 所有查詢都綁定 user_id，從資料層就杜絕越權
export const noteRepo = {
  list(userId, q = '') {
    if (q) {
      const like = `%${q}%`;
      return db
        .prepare(
          'SELECT * FROM notes WHERE user_id = ? AND (title LIKE ? OR content LIKE ?) ORDER BY id DESC'
        )
        .all(userId, like, like);
    }
    return db.prepare('SELECT * FROM notes WHERE user_id = ? ORDER BY id DESC').all(userId);
  },
  find(id, userId) {
    return db.prepare('SELECT * FROM notes WHERE id = ? AND user_id = ?').get(id, userId);
  },
  create(userId, { title, content = '' }) {
    const info = db
      .prepare('INSERT INTO notes (user_id, title, content) VALUES (?, ?, ?)')
      .run(userId, title, content);
    return this.find(info.lastInsertRowid, userId);
  },
  update(id, userId, { title, content }) {
    db.prepare(
      "UPDATE notes SET title = ?, content = ?, updated_at = datetime('now') WHERE id = ? AND user_id = ?"
    ).run(title, content, id, userId);
    return this.find(id, userId);
  },
  remove(id, userId) {
    return db.prepare('DELETE FROM notes WHERE id = ? AND user_id = ?').run(id, userId).changes;
  },
};
```

## `src/middleware/error.js`

```js
export class AppError extends Error {
  constructor(message, statusCode = 500) {
    super(message);
    this.statusCode = statusCode;
  }
}

// 包裝 async handler，丟出的錯誤自動進錯誤 middleware
export const wrap = (fn) => (req, res, next) =>
  Promise.resolve(fn(req, res, next)).catch(next);

// 統一錯誤處理 middleware（4 個參數！放最後）
export function errorHandler(err, req, res, next) {
  const status = err.statusCode || 500;
  if (status === 500) console.error(err);
  res.status(status).json({ error: err.message || '伺服器錯誤' });
}
```

## `src/middleware/auth.js`

```js
import jwt from 'jsonwebtoken';
import { AppError } from './error.js';

export function authMiddleware(req, res, next) {
  const header = req.headers.authorization || '';
  const token = header.startsWith('Bearer ') ? header.slice(7) : null;
  if (!token) return next(new AppError('未提供認證 token', 401));
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    next(new AppError('token 無效或已過期', 401));
  }
}
```

## `src/routes/auth.js`

```js
import { Router } from 'express';
import bcrypt from 'bcryptjs';
import jwt from 'jsonwebtoken';
import { userRepo } from '../repos/userRepo.js';
import { AppError, wrap } from '../middleware/error.js';

const router = Router();

router.post(
  '/register',
  wrap(async (req, res) => {
    const { email, password } = req.body;
    if (!email || !password) throw new AppError('email 與 password 為必填', 400);
    if (userRepo.findByEmail(email)) throw new AppError('email 已被註冊', 400);
    const passwordHash = await bcrypt.hash(password, 10);
    const user = userRepo.create(email, passwordHash);
    res.status(201).json({ id: user.id, email: user.email });
  })
);

router.post(
  '/login',
  wrap(async (req, res) => {
    const { email, password } = req.body;
    const user = userRepo.findByEmail(email);
    // 帳號錯與密碼錯都回一樣訊息，避免洩漏「帳號是否存在」
    if (!user || !(await bcrypt.compare(password, user.password_hash))) {
      throw new AppError('帳號或密碼錯誤', 401);
    }
    const token = jwt.sign({ sub: user.id, email: user.email }, process.env.JWT_SECRET, {
      expiresIn: '1h',
    });
    res.json({ token });
  })
);

export default router;
```

## `src/routes/notes.js`

```js
import { Router } from 'express';
import { noteRepo } from '../repos/noteRepo.js';
import { authMiddleware } from '../middleware/auth.js';
import { AppError, wrap } from '../middleware/error.js';

const router = Router();
router.use(authMiddleware); // 整組 /notes 都需登入

router.get('/', wrap(async (req, res) => {
  res.json(noteRepo.list(req.user.sub, req.query.q || ''));
}));

router.post('/', wrap(async (req, res) => {
  const { title, content } = req.body;
  if (typeof title !== 'string' || title.trim() === '') {
    throw new AppError('title 為必填且需為非空字串', 400);
  }
  res.status(201).json(noteRepo.create(req.user.sub, { title: title.trim(), content }));
}));

router.get('/:id', wrap(async (req, res) => {
  const note = noteRepo.find(Number(req.params.id), req.user.sub);
  if (!note) throw new AppError('找不到筆記', 404); // 別人的筆記也回 404，不洩漏存在性
  res.json(note);
}));

router.put('/:id', wrap(async (req, res) => {
  const existing = noteRepo.find(Number(req.params.id), req.user.sub);
  if (!existing) throw new AppError('找不到筆記', 404);
  const { title, content } = req.body;
  if (typeof title !== 'string' || title.trim() === '') {
    throw new AppError('title 為必填且需為非空字串', 400);
  }
  res.json(
    noteRepo.update(existing.id, req.user.sub, {
      title: title.trim(),
      content: content ?? existing.content,
    })
  );
}));

router.delete('/:id', wrap(async (req, res) => {
  const changes = noteRepo.remove(Number(req.params.id), req.user.sub);
  if (!changes) throw new AppError('找不到筆記', 404);
  res.status(204).end();
}));

export default router;
```

## `src/app.js`

```js
import express from 'express';
import authRoutes from './routes/auth.js';
import noteRoutes from './routes/notes.js';
import { AppError, errorHandler } from './middleware/error.js';

export function createApp() {
  const app = express();
  app.use(express.json());

  app.use('/auth', authRoutes);
  app.use('/notes', noteRoutes);

  app.use((req, res, next) => next(new AppError('路由不存在', 404)));
  app.use(errorHandler); // 4 參數，放最後

  return app;
}
```

## `tests/notes.test.js`（節錄：權限隔離與錯誤路徑）

```js
import request from 'supertest';
import { createApp } from '../src/app.js';

const app = createApp();

async function register(email, password) {
  await request(app).post('/auth/register').send({ email, password });
  const res = await request(app).post('/auth/login').send({ email, password });
  return res.body.token;
}

describe('Notes API 權限隔離', () => {
  test('未帶 token 存取 /notes 回 401', async () => {
    const res = await request(app).get('/notes');
    expect(res.status).toBe(401);
  });

  test('A 不能讀 B 的筆記（回 404）', async () => {
    const tokenA = await register('a@example.com', 'pw12345');
    const tokenB = await register('b@example.com', 'pw12345');

    const created = await request(app)
      .post('/notes')
      .set('Authorization', `Bearer ${tokenA}`)
      .send({ title: 'A 的秘密' });

    const res = await request(app)
      .get(`/notes/${created.body.id}`)
      .set('Authorization', `Bearer ${tokenB}`);

    expect(res.status).toBe(404); // 不可回 200 或 403 洩漏存在
  });

  test('登入密碼錯回 401', async () => {
    await register('c@example.com', 'pw12345');
    const res = await request(app).post('/auth/login').send({ email: 'c@example.com', password: 'wrong' });
    expect(res.status).toBe(401);
  });
});
```

## 關鍵實作片段：權限隔離（最常被忽略、也最能展現後端思維）

```js
// noteRepo.js — 所有查詢都綁定 user_id，從資料層就杜絕越權
export function findNote(db, id, userId) {
  return db.prepare('SELECT * FROM notes WHERE id = ? AND user_id = ?').get(id, userId);
}

// routes/notes.js
router.get('/:id', authMiddleware, wrap(async (req, res) => {
  const note = findNote(db, Number(req.params.id), req.user.sub);
  if (!note) throw new AppError('找不到筆記', 404); // 別人的筆記也回 404，不洩漏存在性
  res.json(note);

```

## 重點

- **權限隔離**：每個查詢都帶 `WHERE user_id = req.user.sub`，從資料層杜絕越權；存取他人資源一律回 **404**（不洩漏存在性）。
- **密碼安全**：bcrypt 雜湊存、登入訊息不區分帳號/密碼錯。
- **機密管理**：`JWT_SECRET` 等全在 `.env`，啟動時檢查必要變數。
- **測試覆蓋**：成功路徑 + 失敗路徑（401/403/404/400）都要測。
